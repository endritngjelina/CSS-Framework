FlexCore is a utility-first CSS framework designed for developers who need to build responsive, accessible, and high-performance interfaces with minimal effort.

app.js
// Create responsive card with FlexCore
import React from 'react';
import 'flexcore/dist/flexcore.min.css';
function ProductCard({ product }) {
  return (
    <div className="bg-white rounded-xl shadow-md overflow-hidden">
      <img src={product.image} alt={product.name} 
           className="w-full h-48 object-cover"/>
      <div className="p-6">
        <div className="font-bold text-xl mb-2">{product.name}</div>
        <p className="text-secondary-600 text-base">
          {product.description}
        </p>
      </div>
      <div className="px-6 py-4 bg-secondary-50 
                     flex justify-between items-center">
        <span className="font-bold text-primary-600">${product.price}</span>
        <button className="bg-primary-500 text-white 
                          px-4 py-2 rounded-lg hover:bg-primary-600">
          Add to Cart
        </button>
      </div>
    </div>
  );
}
